ALTER PROCEDURE [dbo].[SP_CHECKMALOP]
      @MALOP char(8)
AS
BEGIN
      SET NOCOUNT ON;
      IF EXISTS(SELECT MALOP=@MALOP
                        FROM dbo.LOP
                        WHERE  MALOP= @MALOP) return 1;
	ELSE IF EXISTS(SELECT 1 FROM LINK1.QLDSV.dbo.LOP where MALOP=@MALOP) RETURN 1;
	ELSE
		RETURN 0;
END
